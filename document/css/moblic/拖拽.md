## 拖拽

### 思路
1. 获取手指距离浏览器左边距离，元素距离浏览器左边距离
2. 顶部相同；
3. 获取手指距离元素，顶部，左部距离；
4. touchmove实时运动获取，手指距离浏览器距离；


```源码js
// 手指按下，给document绑定去除默认事件
const app=document.querySelector('#app');
let   offsetL=0,     // 距离元素左边距离
      offsetT=0 ;     // 距离元素顶部
app.addEventListener('touchstart',function(ev){
    const touch=ev.touches[0];
    offsetL=touch.clientX-app.offsetLeft;
    offsetT=touch.clientY-app.offsetTop;

    document.addEventListener('touchmove',defaultMover,false);
},false);

app.addEventListener('touchmove',function(ev){
    const touch=ev.touches[0];
    let oL=touch.clientX-offsetL,
        oT=touch.clientY-offsetT;

        if(oT<0){
            return;
        }
        // app.style.left=oL+'px';
        app.style.top=oT+'px';

},false);

app.addEventListener('touchend',function(ev){
    document.removeEventListener('touchmove',defaultMover,false);
},false);

function defaultMover(ev){
    ev.preventDefault();
}
```
