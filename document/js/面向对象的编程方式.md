```
//prototype写面向对象的一种语法糖 

class Person{
    // 构造函数
    constructor(name,age){
        this.name=name;
        this.age=age;
    }

    sayHello(){
        console.log(this.name+this.age);
    }
}

new Person('jmz',24).sayHello();

class SubPerson extends Person{
    constructor(name,age,sex){
        super(name,age);
        this.sex=sex;
    }

    // 重写父类的方法
    sayHello(){
        console.log(this.name+this.age+this.sex);
    }
}

new SubPerson('jmz',24,'男').sayHello();
```
